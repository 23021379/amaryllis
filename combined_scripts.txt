import os
import logging
import numpy as np
import geopandas as gpd
import pandas as pd
from pathlib import Path

# --- CONFIGURATION ---
BASE_DATA_DIR = Path(r"C:\Users\brand\Desktop\renewables\NG_data\artifacts")
L1_PLACES_INPUT = BASE_DATA_DIR / "L1_osm_places.gpkg"
TARGET_CRS = "EPSG:27700"
NULL_SENTINEL = -1.0

# --- MODULE-LEVEL STATE ---
gdf_major_places = None

def _initialize_module_state():
    """Loads and filters the L1 Places artifact once per worker."""
    global gdf_major_places
    if gdf_major_places is not None: return

    try:
        logging.info("Initializing module state for Proximity Places Executor...")
        gdf_places = gpd.read_file(L1_PLACES_INPUT)
        major_places_filter = gdf_places['super_class'].isin(['city', 'town', 'national_capital'])
        gdf_major_places = gdf_places[major_places_filter].copy()
        if gdf_major_places.empty: raise ValueError("No major places found in L1 artifact.")
        logging.info(f"Proximity Places Executor initialized with {len(gdf_major_places)} major places.")
    except Exception as e:
        logging.error(f"FATAL: Could not initialize Proximity Places Executor state: {e}")
        gdf_major_places = "INIT_FAILED"

def execute(state: dict) -> dict:
    """Calculates the distance to the nearest major city or town using sjoin_nearest."""
    if isinstance(gdf_major_places, str) and gdf_major_places == "INIT_FAILED":
        state['errors'].append("EXEC_023_FAIL: Module state could not be initialized.")
        return state

    try:
        site_gdf = gpd.GeoDataFrame([1], geometry=[state['input_geom']], crs=TARGET_CRS)
        joined = gpd.sjoin_nearest(site_gdf, gdf_major_places, how='left', distance_col='distance_m')
        distance_km = joined['distance_m'].iloc[0] / 1000 if not joined.empty else NULL_SENTINEL
        state['features']['dist_to_nearest_major_city_km'] = distance_km

    except Exception as e:
        state['errors'].append(f"EXEC_023_FAIL: {e}")
        state['features']['dist_to_nearest_major_city_km'] = NULL_SENTINEL
            
    return state

import os
import logging
import geopandas as gpd
from pathlib import Path

# --- CONFIGURATION ---
BASE_DATA_DIR = Path(r"C:\Users\brand\Desktop\renewables\NG_data\artifacts")
L1_LAND_COVER_INPUT = BASE_DATA_DIR / "L1_clc_land_cover.gpkg"
L1_BUILDINGS_INPUT = BASE_DATA_DIR / "L1_osm_buildings.gpkg"
L1_LANDUSE_INPUT = BASE_DATA_DIR / "L1_osm_landuse.gpkg"
RADII_METERS = [1000, 2000, 5000, 10000]

# --- MODULE-LEVEL STATE ---
gdf_lc, gdf_bld, gdf_lu, sindex_lc, sindex_bld, sindex_lu = [None] * 6

def _initialize_module_state():
    """Loads all polygon-based L1 artifacts and builds spatial indexes."""
    global gdf_lc, gdf_bld, gdf_lu, sindex_lc, sindex_bld, sindex_lu
    if sindex_lc is not None: return

    try:
        logging.info("Initializing module state for Land Fabric Executor...")
        gdf_lc, sindex_lc = gpd.read_file(L1_LAND_COVER_INPUT), gpd.read_file(L1_LAND_COVER_INPUT).sindex
        gdf_bld, sindex_bld = gpd.read_file(L1_BUILDINGS_INPUT), gpd.read_file(L1_BUILDINGS_INPUT).sindex
        gdf_lu, sindex_lu = gpd.read_file(L1_LANDUSE_INPUT), gpd.read_file(L1_LANDUSE_INPUT).sindex
        logging.info("Land Fabric Executor initialized successfully.")
    except Exception as e:
        logging.error(f"FATAL: Could not initialize Land Fabric Executor state: {e}")
        sindex_lc = "INIT_FAILED"

def execute(state: dict) -> dict:
    if sindex_lc == "INIT_FAILED":
        state['errors'].append("EXEC_024_FAIL: Module state initialized failed.")
        return state

    try:
        for r_m in RADII_METERS:
            r_km = r_m // 1000
            buffer = gpd.GeoDataFrame(geometry=[state['input_geom'].buffer(r_m)], crs="EPSG:27700")
            
            # Helper for indexed overlay
            def get_coverage(gdf, sindex, super_class):
                candidates = gdf.iloc[list(sindex.intersection(buffer.bounds))]
                intersected = gpd.overlay(buffer, candidates[candidates['super_class'] == super_class], how='intersection')
                return intersected.area.sum() / buffer.area.iloc[0] if not intersected.empty else 0.0

            # Calculate features
            state['features'][f'pct_area_urban_fabric_{r_km}km'] = get_coverage(gdf_lc, sindex_lc, 'urban_fabric')
            state['features'][f'pct_area_agricultural_{r_km}km'] = get_coverage(gdf_lc, sindex_lc, 'agricultural')
            state['features'][f'pct_area_protected_natural_{r_km}km'] = get_coverage(gdf_lc, sindex_lc, 'natural_protected_clc')
            state['features'][f'pct_area_industrial_commercial_{r_km}km'] = get_coverage(gdf_lu, sindex_lu, 'industrial_commercial')
            
            # Building density is slightly different (area of buildings / area of buffer in sqkm)
            bld_candidates = gdf_bld.iloc[list(sindex_bld.intersection(buffer.bounds))]
            res_bld = bld_candidates[bld_candidates['super_class'] == 'residential_building']
            bld_intersected = gpd.overlay(buffer, res_bld, how='intersection')
            state['features'][f'residential_building_density_{r_km}km_sqkm'] = bld_intersected.area.sum() / (buffer.area.iloc[0] / 1e6) if not bld_intersected.empty else 0.0

    except Exception as e:
        state['errors'].append(f"EXEC_024_FAIL: {e}")
            
    return state

import geopandas as gpd
import numpy as np
from scipy.spatial import cKDTree
import logging
import os

# --- Executor Configuration ---
BASE_DATA_DIR = r"C:\Users\brand\Desktop\renewables" # As defined in monolith
SUBSTATION_L1_ARTIFACT = os.path.join(BASE_DATA_DIR, "Substations_Amaryllis_L1.gpkg") # Authoritative L1 artifact
PROJECT_CRS = "EPSG:27700"
K_NEIGHBORS = 5
NULL_SENTINEL = -1.0 # Use float for consistency

# --- Module-level State for Performance ---
# These variables are initialized once per worker process, not per function call.
# This prevents catastrophic I/O and indexing overhead.
substation_gdf = None
substation_tree = None
forecast_years = None

def _initialize_module_state():
    """
    Loads the L1 artifact and builds the spatial index.
    This is called only once when the first point is processed by a worker.
    """
    global substation_gdf, substation_tree, forecast_years
    try:
        logging.info(f"Initializing module state for DFES Headroom Executor...")
        substation_gdf = gpd.read_file(SUBSTATION_L1_ARTIFACT)
        
        # PRE-OPERATIVE CHECK: CRS VALIDATION (Pattern 1)
        assert substation_gdf.crs.to_string() == PROJECT_CRS, f"FATAL: CRS Mismatch in L1 artifact. Expected {PROJECT_CRS}, found {substation_gdf.crs}."
        
        # Build spatial index
        substation_coords = np.array(list(zip(substation_gdf.geometry.x, substation_gdf.geometry.y)))
        substation_tree = cKDTree(substation_coords)
        
        # Pre-calculate available forecast years from artifact columns
        forecast_years = sorted([int(col.split('_')[-1]) for col in substation_gdf.columns if 'headroom_mw_' in col])
        
        logging.info(f"DFES Headroom Executor initialized successfully. {len(substation_gdf)} substations indexed.")

    except Exception as e:
        logging.error(f"FATAL: Could not initialize DFES Headroom Executor state: {e}")
        # Setting state to non-None but invalid values ensures subsequent calls fail cleanly.
        substation_gdf = gpd.GeoDataFrame()
        substation_tree = None
        forecast_years = []

def _get_relevant_headroom_col(submission_year: int) -> str:
    """Selects the correct headroom forecast column based on the submission year."""
    for forecast_year in forecast_years:
        if submission_year < forecast_year:
            return f'headroom_mw_{forecast_year}'
    # Fallback to the latest available forecast if submission year is beyond all forecasts
    return f'headroom_mw_{forecast_years[-1]}'

def execute(state: dict) -> dict:
    """
    Calculates grid capacity and proximity features based on DFES substation data.
    
    Adds the following to state['features']:
    - dist_to_nearest_sub_km
    - nearest_sub_voltage_kv
    - nearest_sub_headroom_mw
    - avg_dist_to_5nn_sub_km
    - avg_headroom_5nn_sub_mw
    - std_headroom_5nn_sub_mw
    """
    # Lazy initialization: Load data only when the first point is processed.
    if substation_tree is None:
        _initialize_module_state()
        if substation_tree is None: # Check if initialization failed
            state['errors'].append("EXEC_001_FAIL: Module state could not be initialized.")
            return state

    try:
        input_geom = state['input_geom']
        submission_year = state['submission_year']
        
        target_coord = [input_geom.x, input_geom.y]
        distances_m, indices = substation_tree.query(target_coord, k=K_NEIGHBORS)
        
        if np.isinf(distances_m).any():
             state['errors'].append("EXEC_001_FAIL: Could not find neighbors; cKDTree query failed.")
             return state
             
        neighbors = substation_gdf.iloc[indices]
        relevant_headroom_col = _get_relevant_headroom_col(submission_year)
        
        # --- Feature Synthesis ---
        # Nearest Substation (n=1) features
        nearest_neighbor = neighbors.iloc[0]
        state['features']['dist_to_nearest_sub_km'] = distances_m[0] / 1000.0
        state['features']['nearest_sub_voltage_kv'] = nearest_neighbor.get('voltage', NULL_SENTINEL)
        state['features']['nearest_sub_headroom_mw'] = nearest_neighbor.get(relevant_headroom_col, NULL_SENTINEL)
        
        # Substation Cluster (n=k) features
        state['features']['avg_dist_to_5nn_sub_km'] = np.mean(distances_m) / 1000.0
        
        if relevant_headroom_col in neighbors.columns:
            headroom_values = neighbors[relevant_headroom_col].dropna()
            if not headroom_values.empty:
                state['features']['avg_headroom_5nn_sub_mw'] = headroom_values.mean()
                state['features']['std_headroom_5nn_sub_mw'] = headroom_values.std() if len(headroom_values) > 1 else 0.0
            else:
                state['features']['avg_headroom_5nn_sub_mw'] = NULL_SENTINEL
                state['features']['std_headroom_5nn_sub_mw'] = NULL_SENTINEL
        else:
             state['features']['avg_headroom_5nn_sub_mw'] = NULL_SENTINEL
             state['features']['std_headroom_5nn_sub_mw'] = NULL_SENTINEL
             
    except Exception as e:
        error_msg = f"EXEC_001_FAIL: Unhandled exception during feature synthesis: {e}"
        logging.debug(error_msg) # Log full trace for debugging if needed
        state['errors'].append(error_msg)
        # Ensure feature keys exist even on failure to maintain consistent schema
        feature_keys = [
            'dist_to_nearest_sub_km', 'nearest_sub_voltage_kv', 'nearest_sub_headroom_mw',
            'avg_dist_to_5nn_sub_km', 'avg_headroom_5nn_sub_mw', 'std_headroom_5nn_sub_mw'
        ]
        for key in feature_keys:
            if key not in state['features']:
                state['features'][key] = NULL_SENTINEL

    return state

NULL_SENTINEL = -1.0

# --- Module-level State for Performance ---

fault_tree = None

def _initialize_module_state():
    """Loads the L1 fault level artifact and builds the spatial index once per worker."""
    global gdf_fault_profiles, fault_tree
    try:
        logging.info(f"Initializing module state for Power Quality Executor...")
        gdf_fault_profiles = gpd.read_file(FAULT_LEVEL_L1_ARTIFACT)

        gdf_ohl_lines = gdf_master[gdf_master['voltage'] == '33kv'].copy()
        sindex_ohl_lines = gdf_ohl_lines.sindex
        
        # Generate corridor polygons on-the-fly from the canonical source
        gdf_ohl_corridors = gdf_ohl_lines.copy()
        gdf_ohl_corridors['geometry'] = gdf_ohl_corridors.apply(lambda row: row.geometry.buffer(row.buffer_m), axis=1)
        sindex_ohl_corridors = gdf_ohl_corridors.sindex
        
        logging.info("OHL 33kV Executor initialized successfully.")

    except Exception as e:
        logging.error(f"FATAL: Could not initialize OHL 33kV Executor state: {e}")
        sindex_ohl_lines = "INIT_FAILED"

def execute(state: dict) -> dict:
    """Calculates features related to 33kV OHL proximity, density, and topology."""
    if sindex_ohl_lines is None:
        _initialize_module_state()
    if sindex_ohl_lines == "INIT_FAILED":
        state['errors'].append("EXEC_012_FAIL: Module state could not be initialized.")
        return state

    feature_keys = [
        'ohl_33kv_dist_to_nearest_corridor_m', 'ohl_33kv_point_is_within_corridor',
        'ohl_33kv_density_in_2km_km_per_km2', 'ohl_33kv_nearest_primary_substation_dist_ratio'
    ]

    try:
        geom = state['input_geom']
        
        # --- Proximity & Sterilization Features (using Corridors) ---
        # Find if the point itself is within a corridor
        possible_matches_idx = list(sindex_ohl_corridors.intersection(geom.bounds))
        containing_corridors = gdf_ohl_corridors.iloc[possible_matches_idx]
        is_within = containing_corridors.intersects(geom).any()
        
        state['features']['ohl_33kv_point_is_within_corridor'] = 1 if is_within else 0
        
        # Calculate distance to nearest corridor boundary
        if is_within:
            state['features']['ohl_33kv_dist_to_nearest_corridor_m'] = 0.0
        else:
            # Use the more robust sindex.nearest for finding the closest geometry
            _, tree_indices = sindex_ohl_corridors.nearest(geom, return_all=False)
            nearest_corridor = gdf_ohl_corridors.geometry.iloc[tree_indices[0]]
            state['features']['ohl_33kv_dist_to_nearest_corridor_m'] = geom.distance(nearest_corridor)

        # --- Network Topology & Density Features (using Lines) ---
        buffer = geom.buffer(DENSITY_BUFFER_KM * 1000)
        buffer_area_km2 = np.pi * (DENSITY_BUFFER_KM ** 2)
        
        candidate_lines_idx = list(sindex_ohl_lines.intersection(buffer.bounds))
        if candidate_lines_idx:
            candidate_lines = gdf_ohl_lines.iloc[candidate_lines_idx]
            # Clip the candidate lines to the buffer to get the precise length inside
            clipped_lines = gpd.clip(candidate_lines, buffer)
            total_length_km = clipped_lines.geometry.length.sum() / 1000
            state['features']['ohl_33kv_density_in_2km_km_per_km2'] = total_length_km / buffer_area_km2
        else:
            state['features']['ohl_33kv_density_in_2km_km_per_km2'] = 0.0

        # --- Final Interaction Feature ---
        dist_to_sub_km = state['features'].get('dist_to_nearest_substation_km', 0)
        dist_to_ohl_m = state['features']['ohl_33kv_dist_to_nearest_corridor_m']
        
        if dist_to_sub_km > 0:
            state['features']['ohl_33kv_nearest_primary_substation_dist_ratio'] = dist_to_ohl_m / (dist_to_sub_km * 1000)
        else:
            state['features']['ohl_33kv_nearest_primary_substation_dist_ratio'] = NULL_SENTINEL

    except Exception as e:
        error_msg = f"EXEC_012_FAIL: Unhandled exception during OHL 33kV feature synthesis: {e}"
        logging.debug(error_msg)
        state['errors'].append(error_msg)
        for key in feature_keys:
            if key not in state['features']:
                state['features'][key] = NULL_SENTINEL
                
    return state

def execute(state: dict) -> dict:

        _initialize_module_state()
    if sindex_service_areas == "INIT_FAILED":
        state['errors'].append("EXEC_010_FAIL: Module state could not be initialized.")
        return state

    feature_keys = [
        'substation_area_join_method_within', 'dist_to_governing_substation_centroid_km',
        'forecast_vs_empirical_headroom_delta_pct', 'governing_substation_id_mismatch'

                governing_area = containing_areas.iloc[0]
                join_method = 'within'
        
        # Stage 2: Fallback Proximity Query ('nearest') if Stage 1 failed
        if governing_area is None:
            _, tree_indices = sindex_service_areas.nearest(geom, return_all=False)
            governing_area = gdf_service_areas.iloc[tree_indices[0]]

        state['errors'].append("EXEC_011_FAIL: Module state could not be initialized.")

        'sec_sub_gmt_ratio_knn50', 'sec_sub_avg_customer_count_knn50', 'sec_sub_avg_utilisation_knn50',

        geom = state['input_geom']

        distance = geom.distance(nearest_woodland.geometry)

        possible_matches_idx = list(sindex_woodlands.intersection(buffer.bounds))
        candidates = gdf_woodlands.iloc[possible_matches_idx].copy()

        logging.debug(error_msg)
        state['errors'].append(error_msg)
        # Populate nulls for all expected keys on failure
        
    return state

import logging

PROJECT_CRS = "EPSG:27700"

NULL_SENTINEL_FLOAT = 0.0
NULL_SENTINEL_INT = 0

# --- Module-level State for Performance ---
gdf_np, sindex_np = None, None
gdf_lpa, sindex_lpa = None, None

def _initialize_module_state():
    """Loads L1 artifacts and pre-calculates expensive lookups once per worker."""
    global gdf_np, sindex_np, gdf_lpa, sindex_lpa
    if gdf_np is not None:
        return

    try:
        logging.info("Initializing module state for National Park Executor...")
        
        # 1. Load National Parks data
        gdf_np_local = gpd.read_file(NP_L1_ARTIFACT)
        # PRE-OPERATIVE CHECK: CRS VALIDATION (Pattern 1)
        assert gdf_np_local.crs.to_string() == PROJECT_CRS, "NP L1 Artifact CRS mismatch."
        gdf_np_local['np_area_sqkm'] = gdf_np_local.geometry.area / 1_000_000

        # 2. Load LPA data and perform one-time coverage calculation
        gdf_lpa_local = gpd.read_file(LPA_L1_ARTIFACT)
        if gdf_lpa_local.crs.to_string() != PROJECT_CRS:
            gdf_lpa_local = gdf_lpa_local.to_crs(PROJECT_CRS)
        
        gdf_lpa_local['lpa_area_sqkm'] = gdf_lpa_local.geometry.area / 1_000_000
        np_lpa_intersection = gpd.overlay(gdf_lpa_local, gdf_np_local, how='intersection', keep_geom_type=False)
        np_lpa_intersection['intersection_area_sqkm'] = np_lpa_intersection.geometry.area / 1_000_000
        lpa_coverage = np_lpa_intersection.groupby('LPA23NM')['intersection_area_sqkm'].sum()
        
        gdf_lpa_local = gdf_lpa_local.join(lpa_coverage.rename('np_coverage_sqkm'), on='LPA23NM')
        gdf_lpa_local['lpa_np_coverage_pct'] = (gdf_lpa_local['np_coverage_sqkm'] / gdf_lpa_local['lpa_area_sqkm'] * 100).fillna(0)
        
        # Set final module state atomically
        gdf_np = gdf_np_local
        sindex_np = gdf_np.sindex
        gdf_lpa = gdf_lpa_local
        sindex_lpa = gdf_lpa.sindex

        logging.info("National Park Executor initialized successfully.")

    except Exception as e:
        logging.error(f"FATAL: Could not initialize National Park Executor state: {e}")
        # Mark as failed to prevent execution
        sindex_np = "INIT_FAILED"

def execute(state: dict) -> dict:
    """Calculates National Park features and updates environmental synthesis."""
    if sindex_np is None:
        _initialize_module_state()
    if sindex_np == "INIT_FAILED":
        state['errors'].append("EXEC_NP_FAIL: Module state could not be initialized.")
        return state

    try:
        geom = state['input_geom']
        features = state['features']

        # --- Proximity & Direct Impact Features ---
        nearest_idx = sindex_np.nearest(geom, return_all=False)[1][0]
        nearest_np = gdf_np.iloc[nearest_idx]
        
        features['np_dist_to_nearest_m'] = geom.distance(nearest_np.geometry)
        features['np_nearest_name'] = nearest_np.get('name', 'UNKNOWN')
        features['np_nearest_area_sqkm'] = nearest_np.get('np_area_sqkm', NULL_SENTINEL_FLOAT)
        features['np_is_within'] = 1 if geom.within(nearest_np.geometry) else 0

        # --- Multi-Radii Density (Surgical Intersection Method) ---
        max_radius = max(RADII_METERS)
        buffer = geom.buffer(max_radius)
        possible_matches_idx = list(sindex_np.intersection(buffer.bounds))
        candidates = gdf_np.iloc[possible_matches_idx]
        
        for r in RADII_METERS:
            r_km = r // 1000
            radius_buffer = geom.buffer(r)
            intersecting_nps = candidates[candidates.intersects(radius_buffer)]
            
            total_area_in_radius = 0
            if not intersecting_nps.empty:
                for _, np_geom in intersecting_nps.geometry.items():
                    total_area_in_radius += radius_buffer.intersection(np_geom).area
            
            features[f'np_total_area_in_{r_km}km_sqkm'] = total_area_in_radius / 1_000_000
            features[f'np_count_in_{r_km}km'] = len(intersecting_nps)

        # --- LPA Context Feature (Fast Lookup) ---
        lpa_idx = sindex_lpa.query(geom, predicate='intersects')[0]
        if lpa_idx.size > 0:
            governing_lpa = gdf_lpa.iloc[lpa_idx[0]]
            features['lpa_np_coverage_pct'] = governing_lpa.get('lpa_np_coverage_pct', NULL_SENTINEL_FLOAT)
        else:
            features['lpa_np_coverage_pct'] = NULL_SENTINEL_FLOAT

        # --- Capstone Synthesis Update ---
        constraints = {
            'AncientWoodland': features.get('aw_dist_to_nearest_m', np.inf),
            'AONB': features.get('aonb_dist_to_nearest_m', np.inf),
            'SSSI': features.get('sssi_dist_to_nearest_m', np.inf),
            'NationalPark': features.get('np_dist_to_nearest_m', np.inf)
        }
        min_dist = min(constraints.values())
        features['env_min_dist_to_any_constraint_m'] = min_dist if min_dist != np.inf else -1.0
        if min_dist == np.inf:
            features['env_dominant_constraint_type'] = 'None'
        else:
            features['env_dominant_constraint_type'] = min(constraints, key=constraints.get)

    except Exception as e:
        error_msg = f"EXEC_NP_FAIL: Unhandled exception: {e}"
        logging.debug(error_msg)
        state['errors'].append(error_msg)

    return state

BASE_DATA_DIR = r"C:\Users\brand\Desktop\renewables"
NHLC_SOURCE_ARTIFACT = os.path.join(BASE_DATA_DIR, "NHLC_Amaryllis_L1_Sanitized.gpkg")
PROJECT_CRS = "EPSG:27700"
RADII_METERS = [2000, 5000, 10000]
URBAN_TYPES = ['SETTLEMENT', 'COMMERCE', 'INDUSTRY', 'CIVIC PROVISION']
NULL_SENTINEL_FLOAT = 0.0
NULL_SENTINEL_INT = 0

# --- Module-level State for Performance ---
gdf_urban, sindex_urban = None, None
gdf_settlement, sindex_settlement = None, None

def _initialize_module_state():
    """Performs one-time ingestion, sanitization, and triage of the raw NHLC data."""
    global gdf_urban, sindex_urban, gdf_settlement, sindex_settlement
    if gdf_urban is not None:
        return

    try:
        logging.info("Initializing module state for NHLC Executor (performing one-time triage)...")
        
        # Ingest raw data
        nhlc_gdf = gpd.read_file(NHLC_SOURCE_ARTIFACT)
        assert nhlc_gdf.crs.to_string() == PROJECT_CRS, "NHLC Source Artifact CRS mismatch."
        
        # Sanitize
        nhlc_gdf.columns = nhlc_gdf.columns.str.lower()
        nhlc_gdf.dropna(subset=['geometry'], inplace=True)
        nhlc_gdf = nhlc_gdf[nhlc_gdf.is_valid]

        # The Surgical Triage: Isolate Urban Proxies and discard the rest
        gdf_urban_local = nhlc_gdf[nhlc_gdf['dominantbroadtype'].isin(URBAN_TYPES)].copy()
        gdf_settlement_local = gdf_urban_local[gdf_urban_local['dominantbroadtype'] == 'SETTLEMENT'].copy()
        
        # Set final module state atomically
        gdf_urban = gdf_urban_local
        sindex_urban = gdf_urban.sindex
        gdf_settlement = gdf_settlement_local
        sindex_settlement = gdf_settlement.sindex

        logging.info(f"NHLC Executor initialized successfully with {len(gdf_urban)} urban polygons.")

    except Exception as e:
        logging.error(f"FATAL: Could not initialize NHLC Executor state: {e}")
        sindex_urban = "INIT_FAILED"

def execute(state: dict) -> dict:
    """Calculates all NHLC-derived urban proxy features."""
    if sindex_urban is None:
        _initialize_module_state()
    if sindex_urban == "INIT_FAILED":
        state['errors'].append("EXEC_NHLC_FAIL: Module state could not be initialized.")
        return state

    try:
        geom = state['input_geom']
        features = state['features']

        # --- Direct Impact Features ---
        intersecting_indices = sindex_urban.query(geom, predicate='intersects')
        if intersecting_indices.size > 0:
            features['nhlc_urban_on_site_bool'] = 1
            intersecting_polys = gdf_urban.iloc[intersecting_indices]
            
            # Use overlay only on the tiny subset for accurate area
            intersection_geo = gpd.overlay(
                gpd.GeoDataFrame(geometry=[geom], crs=PROJECT_CRS),
                intersecting_polys, how='intersection'
            )
            features['nhlc_urban_on_site_pct_area'] = intersection_geo.area.sum() / geom.area if geom.area > 0 else 0
        else:
            features['nhlc_urban_on_site_bool'] = 0
            features['nhlc_urban_on_site_pct_area'] = 0.0

        # --- Risk Gradient Features ---
        for r in RADII_METERS:
            r_km = r // 1000
            buffer_geom = geom.buffer(r)
            buffer_area = buffer_geom.area

            # Calculate for all urban types
            urban_candidates_idx = list(sindex_urban.intersection(buffer_geom.bounds))
            urban_candidates = gdf_urban.iloc[urban_candidates_idx]
            urban_in_buffer = urban_candidates[urban_candidates.intersects(buffer_geom)]
            
            features[f'nhlc_urban_cell_count_{r_km}km'] = len(urban_in_buffer)
            urban_area = sum(buffer_geom.intersection(poly).area for _, poly in urban_in_buffer.geometry.items())
            features[f'nhlc_urban_density_{r_km}km'] = urban_area / buffer_area if buffer_area > 0 else 0.0

            # Calculate for settlement-only
            settlement_candidates_idx = list(sindex_settlement.intersection(buffer_geom.bounds))
            settlement_candidates = gdf_settlement.iloc[settlement_candidates_idx]
            settlement_in_buffer = settlement_candidates[settlement_candidates.intersects(buffer_geom)]
            
            settlement_area = sum(buffer_geom.intersection(poly).area for _, poly in settlement_in_buffer.geometry.items())
            features[f'nhlc_settlement_density_{r_km}km'] = settlement_area / buffer_area if buffer_area > 0 else 0.0

    except Exception as e:
        error_msg = f"EXEC_NHLC_FAIL: Unhandled exception: {e}"
        logging.debug(error_msg)
        state['errors'].append(error_msg)

    return state

import os
import logging
import numpy as np
import geopandas as gpd
from shapely.geometry import LineString, box
from sklearn.neighbors import BallTree

# --- CONFIGURATION ---
BASE_DATA_DIR = r"C:\Users\brand\Desktop\renewables"
L1_MASTER_ASSET_ARTIFACT = os.path.join(BASE_DATA_DIR, r"NG_data\artifacts\NG_L1_Topological_Assets.gpkg")
SUBSTATION_L1_ARTIFACT = os.path.join(BASE_DATA_DIR, "Substations_Amaryllis_L1.gpkg")

TARGET_CRS = "EPSG:27700"
DENSITY_RADII_M = [2000, 5000, 10000, 20000]
MAX_RADIUS_M = max(DENSITY_RADII_M)

# --- MODULE-LEVEL STATE ---
gdf_subs, tree_subs = None, None

def _initialize_module_state():
    """Loads substation L1 artifact and builds a BallTree once per worker process."""
    global gdf_subs, tree_subs
    if tree_subs is not None: return

    try:
        logging.info("Initializing module state for DNO Topology Executor...")
        gdf_subs = gpd.read_file(SUBSTATION_L1_ARTIFACT).to_crs(TARGET_CRS)
        assert gdf_subs.crs.to_string() == TARGET_CRS, "Substation CRS Mismatch"
        coords_subs = np.array(list(zip(gdf_subs.geometry.x, gdf_subs.geometry.y)))
        tree_subs = BallTree(coords_subs, metric='euclidean')
        logging.info("DNO Topology Executor initialized successfully.")
    except Exception as e:
        logging.error(f"FATAL: Could not initialize DNO Topology Executor state: {e}")
        tree_subs = "INIT_FAILED"

def execute(state: dict) -> dict:
    """Calculates DNO/NG asset topology features for a single point using the L1 artifact."""
    if tree_subs is None: _initialize_module_state()
    if tree_subs == "INIT_FAILED":
        state['errors'].append("EXEC_018_FAIL: Module state could not be initialized.")
        return state

    try:
        site_geom = state['input_geom']
        
        # --- SURGICAL QUERY PROTOCOL ---
        bbox = box(
            site_geom.x - MAX_RADIUS_M, site_geom.y - MAX_RADIUS_M,
            site_geom.x + MAX_RADIUS_M, site_geom.y + MAX_RADIUS_M
        )
        local_points = gpd.read_file(L1_MASTER_ASSET_ARTIFACT, layer='points', bbox=bbox)
        local_lines = gpd.read_file(L1_MASTER_ASSET_ARTIFACT, layer='lines', bbox=bbox)

        # --- FEATURE ENGINEERING (POINTS) ---
        if not local_points.empty:
            local_points['distance_m'] = local_points.distance(site_geom)
            for _, row in local_points.loc[local_points.groupby(['voltage', 'asset_tag'])['distance_m'].idxmin()].iterrows():
                state['features'][f"ohl_{row.voltage}_dist_to_nearest_{row.asset_tag}_m"] = row.distance_m

            for radius in DENSITY_RADII_M:
                radius_km_str = f"{int(radius / 1000)}km"
                area_km2 = np.pi * ((radius / 1000) ** 2)
                points_in_buffer = local_points[local_points['distance_m'] <= radius]
                if not points_in_buffer.empty:
                    for (v, t), count in points_in_buffer.groupby(['voltage', 'asset_tag']).size().items():
                        state['features'][f"ohl_{v}_{t}_count_{radius_km_str}"] = count
                        state['features'][f"ohl_{v}_{t}_density_per_km2_{radius_km_str}"] = count / area_km2

        # --- FEATURE ENGINEERING (LINES) ---
        if not local_lines.empty:
            local_lines['corridor_geom'] = local_lines.apply(lambda r: r.geometry.buffer(r.buffer_m), axis=1)
            local_lines_gdf = local_lines.set_geometry('corridor_geom')
            
            local_lines_gdf['distance_to_corridor_m'] = local_lines_gdf.distance(site_geom)
            for _, row in local_lines_gdf.loc[local_lines_gdf.groupby(['voltage', 'asset_tag'])['distance_to_corridor_m'].idxmin()].iterrows():
                state['features'][f"{row.asset_tag}_{row.voltage}_dist_to_nearest_corridor_m"] = row.distance_to_corridor_m
            
            for _, row in local_lines_gdf[local_lines_gdf.intersects(site_geom)].iterrows():
                state['features'][f"{row.asset_tag}_{row.voltage}_corridor_intersects_site"] = 1

            _, idx = tree_subs.query([[site_geom.x, site_geom.y]], k=1)
            nearest_sub_geom = gdf_subs.geometry.iloc[idx[0][0]]
            connection_path = LineString([site_geom, nearest_sub_geom])
            for (v, t), count in local_lines_gdf[local_lines_gdf.intersects(connection_path)].groupby(['voltage', 'asset_tag']).size().items():
                state['features'][f"{t}_{v}_connection_path_intersection_count"] = count

            for radius in DENSITY_RADII_M:
                site_buffer = site_geom.buffer(radius)
                clipped_lines = gpd.overlay(local_lines.set_geometry('geometry'), gpd.GeoDataFrame(geometry=[site_buffer], crs=TARGET_CRS), how='intersection')
                if not clipped_lines.empty:
                    clipped_lines['length_km'] = clipped_lines.geometry.length / 1000
                    for (v, t), length in clipped_lines.groupby(['voltage', 'asset_tag'])['length_km'].sum().items():
                        radius_km_str = f"{int(radius/1000)}km"
                        area_km2 = np.pi * ((radius/1000)**2)
                        state['features'][f"{t}_{v}_density_in_{radius_km_str}_km_per_km2"] = length / area_km2

        # --- CAPSTONE SYNTHESIS ---
        intersection_cols = [v for k, v in state['features'].items() if '_connection_path_intersection_count' in k]
        if intersection_cols:
            state['features']['ohl_total_connection_path_intersections'] = sum(intersection_cols)

    except Exception as e:
        error_msg = f"EXEC_018_FAIL: Unhandled exception during DNO topology synthesis: {e}"
        logging.debug(error_msg)
        state['errors'].append(error_msg)

    return state

import os
import logging
import numpy as np
import geopandas as gpd
from scipy.spatial import cKDTree

# --- CONFIGURATION ---
BASE_DATA_DIR = r"C:\Users\brand\Desktop\renewables"
L1_SUBSTATION_ARTIFACT = os.path.join(BASE_DATA_DIR, "Substations_Amaryllis_L1.gpkg")
TARGET_CRS = "EPSG:27700"
KNN_NEIGHBORS = 5
RADIUS_METERS = 10000
NULL_SENTINEL = -1.0

# --- MODULE-LEVEL STATE ---
gdf_subs, spatial_index = None, None

def _initialize_module_state():
    """Loads the L1 substation artifact and builds a cKDTree once per worker process."""
    global gdf_subs, spatial_index
    if spatial_index is not None: return

    try:
        logging.info("Initializing module state for DFES Headroom Executor...")
        gdf_subs = gpd.read_file(L1_SUBSTATION_ARTIFACT)
        assert gdf_subs.crs.to_string() == TARGET_CRS, "Substation L1 Artifact CRS Mismatch"
        
        coords = np.array(list(zip(gdf_subs.geometry.x, gdf_subs.geometry.y)))
        spatial_index = cKDTree(coords)
        logging.info(f"DFES Headroom Executor initialized with {len(gdf_subs)} substations.")
    except Exception as e:
        logging.error(f"FATAL: Could not initialize DFES Headroom Executor state: {e}")
        spatial_index = "INIT_FAILED"

def execute(state: dict) -> dict:
    """Calculates direct, k-NN, and radius-based substation features for a single point."""
    if spatial_index is None: _initialize_module_state()
    if spatial_index == "INIT_FAILED":
        state['errors'].append("EXEC_001_FAIL: Module state could not be initialized.")
        return state

    try:
        site_geom = state['input_geom']
        site_coord = (site_geom.x, site_geom.y)

        # --- Direct Nearest Neighbor Features ---
        dist_nn1, idx_nn1 = spatial_index.query(site_coord, k=1)
        nearest_sub = gdf_subs.iloc[idx_nn1]
        
        state['features']['dist_to_nearest_sub_km'] = dist_nn1 / 1000
        state['features']['nearest_sub_voltage_kv'] = nearest_sub['max_voltage_kv'].values[0]
        state['features']['nearest_sub_headroom_mw'] = nearest_sub['headroom_mw'].values[0]

        # --- K-Nearest Neighbors Aggregation ---
        distances_knn, indices_knn = spatial_index.query(site_coord, k=KNN_NEIGHBORS)
        neighbor_data = gdf_subs.iloc[indices_knn]

        state['features'][f'avg_dist_to_{KNN_NEIGHBORS}nn_sub_km'] = np.mean(distances_knn) / 1000
        state['features'][f'avg_headroom_{KNN_NEIGHBORS}nn_sub_mw'] = np.nanmean(neighbor_data['headroom_mw'])
        state['features'][f'std_headroom_{KNN_NEIGHBORS}nn_sub_mw'] = np.nanstd(neighbor_data['headroom_mw'])

        # --- Radius-Based Aggregation ---
        indices_radius = spatial_index.query_ball_point(site_coord, r=RADIUS_METERS)
        if indices_radius:
            total_kva_in_radius = gdf_subs.iloc[indices_radius]['total_kva'].sum()
            state['features'][f'total_kva_in_{RADIUS_METERS//1000}km_radius'] = total_kva_in_radius
        else:
            state['features'][f'total_kva_in_{RADIUS_METERS//1000}km_radius'] = 0.0

    except Exception as e:
        error_msg = f"EXEC_001_FAIL: Unhandled exception during substation feature synthesis: {e}"
        logging.debug(error_msg)
        state['errors'].append(error_msg)
        # Ensure feature keys exist to maintain schema consistency
        feature_keys = [
            'dist_to_nearest_sub_km', 'nearest_sub_voltage_kv', 'nearest_sub_headroom_mw',
            f'avg_dist_to_{KNN_NEIGHBORS}nn_sub_km', f'avg_headroom_{KNN_NEIGHBORS}nn_sub_mw',
            f'std_headroom_{KNN_NEIGHBORS}nn_sub_mw', f'total_kva_in_{RADIUS_METERS//1000}km_radius'
        ]
        for key in feature_keys:
            state['features'].setdefault(key, NULL_SENTINEL)
            
    return state

NULL_SENTINEL = -1.0

# --- MODULE-LEVEL STATE ---
gdf_dist_subs, spatial_index = None, None

def _initialize_module_state():
    """Loads the L1 distribution substation artifact and builds a spatial index once per worker."""

    if spatial_index is not None: return

    try:
        logging.info("Initializing module state for Secondary Sites Executor...")

        logging.info(f"Secondary Sites Executor initialized with {len(gdf_dist_subs)} substations.")
    except Exception as e:
        logging.error(f"FATAL: Could not initialize Secondary Sites Executor state: {e}")
        spatial_index = "INIT_FAILED"

def execute(state: dict) -> dict:
    """Calculates direct and aggregated features from the distribution substation network."""
    if spatial_index is None: _initialize_module_state()
    if spatial_index == "INIT_FAILED":
        state['errors'].append("EXEC_019_FAIL: Module state could not be initialized.")
        return state

    try:
        site_geom = state['input_geom']
        site_coord = (site_geom.x, site_geom.y)

        # --- Direct Features (Nearest Secondary Substation) ---
        _, idx_nn1 = spatial_index.query(site_coord, k=1)
        nearest_sec_sub = gdf_dist_subs.iloc[idx_nn1]
        
        state['features']['sec_sub_gov_onan_rating_kva'] = nearest_sec_sub['rating_kva'].values[0]
        state['features']['sec_sub_gov_utilisation_pct'] = nearest_sec_sub['utilisation_pct'].values[0]

        # --- Aggregate Features (Primary Substation Proxy via k-NN) ---
        _, indices_knn = spatial_index.query(site_coord, k=PRIMARY_SUB_PROXY_NEIGHBORS)
        neighbor_data = gdf_dist_subs.iloc[indices_knn]
        
        state['features']['primary_sub_total_kva'] = np.nansum(neighbor_data['rating_kva'])
        state['features']['primary_sub_tx_count'] = PRIMARY_SUB_PROXY_NEIGHBORS
        state['features']['primary_sub_avg_tx_kva'] = np.nanmean(neighbor_data['rating_kva'])
        state['features']['primary_sub_max_tx_kva'] = np.nanmax(neighbor_data['rating_kva'])
        state['features']['primary_sub_min_tx_kva'] = np.nanmin(neighbor_data['rating_kva'])
        state['features']['primary_sub_tx_kva_variance'] = np.nanvar(neighbor_data['rating_kva'])
        state['features']['primary_sub_downstream_avg_utilisation_pct'] = np.nanmean(neighbor_data['utilisation_pct'])
        
        # --- Second-Order Synthesis ---
        # This calculation is now robust as it only depends on features generated within this executor.
        with np.errstate(divide='ignore', invalid='ignore'):
            kva_per_tx = state['features']['primary_sub_total_kva'] / state['features']['primary_sub_tx_count']
            state['features']['kva_per_transformer'] = np.nan_to_num(kva_per_tx, nan=NULL_SENTINEL)

    except Exception as e:
        error_msg = f"EXEC_019_FAIL: Unhandled exception during secondary site synthesis: {e}"
        logging.debug(error_msg)
        state['errors'].append(error_msg)
        # Null-fill all features this executor is responsible for to maintain schema
        feature_keys = [
            'sec_sub_gov_onan_rating_kva', 'sec_sub_gov_utilisation_pct',
            'primary_sub_total_kva', 'primary_sub_tx_count', 'primary_sub_avg_tx_kva',
            'primary_sub_max_tx_kva', 'primary_sub_min_tx_kva', 'primary_sub_tx_kva_variance',
            'primary_sub_downstream_avg_utilisation_pct', 'kva_per_transformer'
        ]
        for key in feature_keys: state['features'].setdefault(key, NULL_SENTINEL)
            
    return state

def clean_col_names(df: pd.DataFrame) -> pd.DataFrame:
    """Amaryllis Standard: Cleans and standardizes DataFrame column names."""
    new_cols = []
    for col in df.columns:
        new_col = col.lower().strip().replace(' ', '_')
        new_col = ''.join(c if c.isalnum() else '_' for c in new_col)
        while '__' in new_col: new_col = new_col.replace('__', '_')
        new_col = new_col.strip('_')
        new_cols.append(new_col)
    df.columns = new_cols
    return df

def aggregate_fault_levels(fault_path: str) -> pd.DataFrame:
    """Phase 1: Ingests, decontaminates, and performs critical max-aggregation."""
    logging.info("Phase 1: Ingesting and aggregating raw fault level data...")
    df = pd.read_csv(fault_path)
    df = clean_col_names(df)

    for col in ['3ph_rms_break', '1ph_rms_break']:
        df[col] = pd.to_numeric(df[col], errors='coerce')

    # CRITICAL MANDATE: Aggregate using max() to model worst-case engineering constraint
    agg_rules = {
        '3ph_rms_break': 'max',
        '1ph_rms_break': 'max'
    }
    df_agg = df.groupby('substation_or_busbar_name').agg(agg_rules).reset_index()
    
    logging.info(f"Aggregated raw fault data from {len(df)} rows to {len(df_agg)} unique substations.")
    return df_agg

def sanitize_join_key(name: str) -> str:
    """Creates a sanitized key for robust direct and fuzzy matching."""
    if not isinstance(name, str): return ''
    name = name.lower()
    # Remove common terms and punctuation
    stopwords = ['kv', 'substation', 'primary', 'bsp', 'grid']
    for word in stopwords:
        name = name.replace(word, '')
    name = re.sub(r'[^a-z0-9]', '', name)
    return name.strip()

def fuse_with_fault_data(gdf_primary: gpd.GeoDataFrame, df_fault: pd.DataFrame) -> gpd.GeoDataFrame:
    """Phase 2: Executes the Fuzzy Geospatial Bridge protocol."""
    logging.info("Phase 2: Executing Fuzzy Geospatial Bridge...")
    
    # Create sanitized join keys in both dataframes
    gdf_primary['sanitized_join_key'] = gdf_primary['name'].apply(sanitize_join_key)
    df_fault['sanitized_join_key'] = df_fault['substation_or_busbar_name'].apply(sanitize_join_key)
    
    # Stage A: Direct Join
    gdf_enriched = pd.merge(
        gdf_primary, df_fault, on='sanitized_join_key', how='left'
    )
    
    # Isolate orphans for Stage B
    orphans = gdf_enriched[gdf_enriched['3ph_rms_break'].isna()].copy()
    successes = gdf_enriched[gdf_enriched['3ph_rms_break'].notna()]
    logging.info(f"Direct join successful for {len(successes)}/{len(gdf_enriched)} records.")
    
    if not orphans.empty:
        logging.info(f"Attempting fuzzy match for {len(orphans)} orphan records...")
        choices = df_fault['sanitized_join_key'].unique()
        
        matches = orphans['sanitized_join_key'].apply(
            lambda x: fuzzy_process.extractOne(x, choices, score_cutoff=FUZZY_MATCH_CONFIDENCE_THRESHOLD)
        )
        
        # Filter out low-confidence matches
        valid_matches = matches.dropna()
        if not valid_matches.empty:
            match_df = pd.DataFrame(valid_matches.tolist(), index=valid_matches.index, columns=['match', 'score'])
            match_df = pd.merge(match_df, df_fault, left_on='match', right_on='sanitized_join_key', how='left')
            
            # Update the orphan records with the matched fault data
            orphans.update(match_df)
            logging.info(f"Successfully matched {len(valid_matches)} records via fuzzy logic.")
        
        # Recombine successes and high-confidence fuzzy matches
        gdf_enriched = pd.concat([successes, orphans], ignore_index=True)

    gdf_enriched.to_parquet(L1_FAULT_ENRICHED_OUTPUT)
    logging.info(f"SUCCESS: Fault-enriched L1 artifact saved to {L1_FAULT_ENRICHED_OUTPUT}")
    return gdf_enriched

def main():

    logging.info("STARTING Directive AD-GRID-12: Fault Level Intelligence Synthesis.")
    try:
        df_fault_agg = aggregate_fault_levels(FAULT_LEVEL_INPUT)
        gdf_primary_subs = gpd.read_parquet(PRIMARY_SUBS_AUTH_INPUT)
        df_solar = pd.read_csv(SOLAR_SITES_INPUT)

    logging.info("Phase 3: Integrating fault features into solar site data...")
    
    # Direct Feature Integration
    gdf_solar_joined = gpd.sjoin_nearest(gdf_solar, gdf_enriched_subs, how="left")
    
    # MANDATED DE-DUPLICATION GUARD RAIL: Immediately cleanse join artifacts.
    if gdf_solar_joined.index.has_duplicates:
        duplicate_count = gdf_solar_joined.index.duplicated().sum()
        logging.warning(f"Detected {duplicate_count} duplicate solar site entries post-join. Enforcing 1-to-1 relationship by keeping first match.")
        gdf_solar_joined = gdf_solar_joined[~gdf_solar_joined.index.duplicated(keep='first')]

    gdf_solar_joined.rename(columns={
        '3ph_rms_break': 'nearest_sub_max_3ph_fault_ka',
        '1ph_rms_break': 'nearest_sub_max_ef_fault_ka'
    }, inplace=True)

    # Aggregate Feature Synthesis (k-NN)
    logging.info(f"Synthesizing {KNN_NEIGHBORS}-NN aggregate fault features...")
    # Ensure we only use substations with valid fault data for aggregation
    gdf_agg_source = gdf_enriched_subs.dropna(subset=['3ph_rms_break', '1ph_rms_break'])
    
    agg_coords = np.array(list(gdf_agg_source.geometry.apply(lambda p: (p.x, p.y))))

NULL_SENTINEL = -1.0

# --- MODULE-LEVEL STATE ---
gdf_lct_events, gdf_primary_subs, spatial_index = None, None, None

def _initialize_module_state():
    """Loads LCT event data and primary substation locations once per worker."""

    if spatial_index is not None: return

    try:
        logging.info("Initializing module state for LCT Executor...")
        gdf_lct_events = gpd.read_file(L2_LCT_SPATIOTEMPORAL_INPUT)

        spatial_index = "INIT_FAILED"

def execute(state: dict) -> dict:
    """Calculates point-in-time LCT saturation features for a single location."""
    if spatial_index is None: _initialize_module_state()
    if spatial_index == "INIT_FAILED":
        state['errors'].append("EXEC_008_FAIL: Module state could not be initialized.")
        return state

    try:
        # --- Temporal Guard Protocol ---
        submission_date = state['date_validated'] # Assumes orchestrator provides this key
        lct_snapshot = gdf_lct_events[gdf_lct_events['connected_month'] <= submission_date]

        if lct_snapshot.empty: # No historical LCT data, null-fill features
            for key in _get_feature_keys(): state['features'][key] = NULL_SENTINEL
            return state

        # --- Dynamic Point-in-Time Aggregation ---
        agg_df = lct_snapshot.groupby('primary_sub_id')[['ev', 'es', 'hp', 'pv']].sum()
        agg_df['total_connections'] = agg_df.sum(axis=1)
        with np.errstate(divide='ignore', invalid='ignore'):
            agg_df['gen_dem_ratio'] = (agg_df['pv'] + agg_df['es']) / (agg_df['ev'] + agg_df['hp'] + 1)
        
        enriched_subs = gdf_primary_subs.merge(agg_df, on='primary_sub_id', how='left').fillna(0)
        
        # --- k-NN Feature Synthesis ---
        site_coord = (state['input_geom'].x, state['input_geom'].y)
        _, indices_knn = spatial_index.query(site_coord, k=KNN_NEIGHBORS)
        
        nearest_subs_data = enriched_subs.iloc[indices_knn]
        nearest_sub = nearest_subs_data.iloc[0]

        # Nearest Neighbor Features (n=1)
        state['features']['lct_nearest_total_connections'] = nearest_sub['total_connections']
        state['features']['lct_nearest_gen_dem_ratio'] = nearest_sub['gen_dem_ratio']
        
        # Local Cluster Features (k-NN)
        state['features'][f'lct_avg_{KNN_NEIGHBORS}nn_total_connections'] = nearest_subs_data['total_connections'].mean()
        state['features'][f'lct_std_{KNN_NEIGHBORS}nn_total_connections'] = nearest_subs_data['total_connections'].std()
        state['features'][f'lct_avg_{KNN_NEIGHBORS}nn_gen_dem_ratio'] = nearest_subs_data['gen_dem_ratio'].mean()

    except Exception as e:
        error_msg = f"EXEC_008_FAIL: Unhandled exception during LCT synthesis: {e}"
        logging.debug(error_msg)
        state['errors'].append(error_msg)
        for key in _get_feature_keys(): state['features'].setdefault(key, NULL_SENTINEL)
            
    return state

def _get_feature_keys():
    """Helper to ensure consistent schema on success or failure."""
    return [
        'lct_nearest_total_connections', 'lct_nearest_gen_dem_ratio',
        f'lct_avg_{KNN_NEIGHBORS}nn_total_connections', f'lct_std_{KNN_NEIGHBORS}nn_total_connections',
        f'lct_avg_{KNN_NEIGHBORS}nn_gen_dem_ratio'
    ]

BASE_DATA_DIR = Path(r"C:\Users\brand\Desktop\renewables\NG_data\artifacts")
L1_LAND_COVER_INPUT = BASE_DATA_DIR / "L1_clc_land_cover.gpkg"
L1_BUILDINGS_INPUT = BASE_DATA_DIR / "L1_osm_buildings.gpkg"
TARGET_CRS = "EPSG:27700"
RADII_METERS = [1000, 2000, 5000, 10000]

# --- MODULE-LEVEL STATE ---
gdf_lc, gdf_bld, sindex_lc, sindex_bld = None, None, None, None

def _initialize_module_state():
    """Loads L1 Land Cover and Buildings artifacts and builds spatial indexes."""
    global gdf_lc, gdf_bld, sindex_lc, sindex_bld
    if sindex_lc is not None: return

    try:
        logging.info("Initializing module state for Land Fabric Executor...")
        gdf_lc = gpd.read_file(L1_LAND_COVER_INPUT)
        sindex_lc = gdf_lc.sindex
        gdf_bld = gpd.read_file(L1_BUILDINGS_INPUT)
        sindex_bld = gdf_bld.sindex
        logging.info("Land Fabric Executor initialized successfully.")
    except Exception as e:
        logging.error(f"FATAL: Could not initialize Land Fabric Executor state: {e}")
        sindex_lc = "INIT_FAILED"

def execute(state: dict) -> dict:
    """Calculates land cover percentages and building density within multiple radii."""
    if sindex_lc == "INIT_FAILED":
        state['errors'].append("EXEC_024_FAIL: Module state could not be initialized.")
        return state

    try:
        geom = state['input_geom']
        for r_m in RADII_METERS:
            r_km = r_m // 1000
            buffer = gpd.GeoDataFrame(geometry=[geom.buffer(r_m)], crs=TARGET_CRS)
            buffer_area_sqkm = buffer.area.iloc[0] / 1e6

            # --- Land Cover Calculations ---
            lc_candidates = gdf_lc.iloc[list(sindex_lc.intersection(buffer.bounds))]
            lc_intersected = gpd.overlay(buffer, lc_candidates, how='intersection')
            if not lc_intersected.empty:
                coverage = lc_intersected.groupby('super_class').area.sum() / buffer.area.iloc[0]
                state['features'][f'pct_area_urban_fabric_{r_km}km'] = coverage.get('urban_fabric', 0.0)
                state['features'][f'pct_area_agricultural_{r_km}km'] = coverage.get('agricultural', 0.0)
                state['features'][f'pct_area_industrial_commercial_{r_km}km'] = coverage.get('industrial_commercial', 0.0)
                state['features'][f'pct_area_protected_natural_{r_km}km'] = coverage.get('natural_protected', 0.0)

            # --- Building Density Calculation ---
            bld_candidates = gdf_bld.iloc[list(sindex_bld.intersection(buffer.bounds))]
            bld_res = bld_candidates[bld_candidates['super_class'] == 'residential_building']
            bld_intersected = gpd.overlay(buffer, bld_res, how='intersection')
            if not bld_intersected.empty:
                 state['features'][f'residential_building_density_{r_km}km_sqkm'] = bld_intersected.area.sum() / buffer_area_sqkm

    except Exception as e:
        state['errors'].append(f"EXEC_024_FAIL: {e}")
            
    return state

import os
import logging
import geopandas as gpd
from pathlib import Path

# --- CONFIGURATION ---
BASE_DATA_DIR = Path(r"C:\Users\brand\Desktop\renewables\NG_data\artifacts")
L1_ROADS_INPUT = BASE_DATA_DIR / "L1_osm_roads.gpkg"
TARGET_CRS = "EPSG:27700"
RADII_METERS = [1000, 2000, 5000, 10000]

# --- MODULE-LEVEL STATE ---
gdf_roads, sindex_roads = None, None

def _initialize_module_state():
    """Loads the L1 Roads artifact and builds a spatial index."""
    global gdf_roads, sindex_roads
    if sindex_roads is not None: return

    try:
        logging.info("Initializing module state for Linear Infrastructure Executor...")
        gdf_roads = gpd.read_file(L1_ROADS_INPUT)
        sindex_roads = gdf_roads.sindex
        logging.info("Linear Infrastructure Executor initialized successfully.")
    except Exception as e:
        logging.error(f"FATAL: Could not initialize Linear Infrastructure Executor state: {e}")
        sindex_roads = "INIT_FAILED"

def execute(state: dict) -> dict:
    """Calculates the length of major roads within multiple radii."""
    if sindex_roads == "INIT_FAILED":
        state['errors'].append("EXEC_025_FAIL: Module state could not be initialized.")
        return state

    try:
        geom = state['input_geom']
        for r_m in RADII_METERS:
            r_km = r_m // 1000
            buffer = gpd.GeoDataFrame(geometry=[geom.buffer(r_m)], crs=TARGET_CRS)
            
            candidates = gdf_roads.iloc[list(sindex_roads.intersection(buffer.bounds))]
            major_roads = candidates[candidates['super_class'] == 'major_road']
            
            if not major_roads.empty:
                clipped_roads = gpd.clip(major_roads, buffer)
                if not clipped_roads.empty:
                    state['features'][f'major_road_length_{r_km}km'] = clipped_roads.length.sum() / 1000 # to km
                else:
                    state['features'][f'major_road_length_{r_km}km'] = 0.0
            else:
                 state['features'][f'major_road_length_{r_km}km'] = 0.0

    except Exception as e:
        state['errors'].append(f"EXEC_025_FAIL: {e}")
            
    return state

import logging

# --- CONFIGURATION ---
NULL_SENTINEL = 0.0

def execute(state: dict) -> dict:
    """Synthesizes Surgical Interaction Constructs (SICs) for the imputation model."""
    try:
        # SIC-IMPUTE-01: "Developed Corridor Intensity"
        # Note: This SIC requires a terrain gradient feature, which is a placeholder.
        # The logic is implemented to match the original script, but will produce 0.
        major_road_len_5km = state['features'].get('major_road_length_5km', 0.0)
        # Placeholder for DEM-derived feature
        stddev_terrain_gradient_5km = state['features'].get('stddev_terrain_gradient_5km', 0.0) 
        state['features']['SIC_Corridor_Intensity'] = major_road_len_5km * stddev_terrain_gradient_5km

        # SIC-IMPUTE-02: "Urban Fringe Demand Pressure"
        urban_fabric_2km = state['features'].get('pct_area_urban_fabric_2km', 0.0)
        agri_fabric_2km = state['features'].get('pct_area_agricultural_2km', 0.0)
        state['features']['SIC_Urban_Fringe_Pressure'] = urban_fabric_2km * agri_fabric_2km

    except Exception as e:
        error_msg = f"EXEC_026_FAIL: Unhandled exception during SIC synthesis: {e}"
        logging.debug(error_msg)
        state['errors'].append(error_msg)
        state['features']['SIC_Corridor_Intensity'] = NULL_SENTINEL
        state['features']['SIC_Urban_Fringe_Pressure'] = NULL_SENTINEL
            
    return state
